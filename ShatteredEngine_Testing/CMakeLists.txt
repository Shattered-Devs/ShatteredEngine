project(ShatteredEngine_Testing)

if(_SHATTERED_WIN32)
    add_executable(ShatteredEngine_Testing WIN32 main.cpp)
elseif(_SHATTERED_WINRT)
    add_executable(ShatteredEngine_Testing WIN32 main.cpp "../libs/UWP/x64/SDL2.dll")
else()
    add_executable(ShatteredEngine_Testing main.cpp)
endif()

# Force C++ 17 Standard
set_property(TARGET ShatteredEngine_Testing PROPERTY CXX_STANDARD 17)

target_include_directories(ShatteredEngine_Testing PUBLIC "../ShatteredEngine" PUBLIC "../3rdParty/termcolor/include" PUBLIC "../3rdParty/SDL/include")

if((_SHATTERED_MSVC OR _SHATTERED_MINGW) AND NOT _SHATTERED_WINRT)
    target_link_libraries(ShatteredEngine_Testing "ShatteredEngine" "SDL2-static" "SDL2main" "bx" "bimg" "bgfx" "gdi32" "kernel32" "psapi")
endif()
if(_SHATTERED_WINRT)
    target_link_directories(ShatteredEngine_Testing PUBLIC "../libs/UWP/x64")
    target_link_libraries(ShatteredEngine_Testing "ShatteredEngine" "SDL2" "gdi32" "kernel32" "psapi")
    set_property(SOURCE "../libs/UWP/x64/SDL2.dll" PROPERTY VS_DEPLOYMENT_CONTENT 1)
endif()
if(_SHATTERED_APPLE)
    set(macOS_Libraries "-framework QuartzCore" "-framework Metal" "-framework IOKit" "-framework CoreVideo" "-framework Foundation")
    target_link_libraries(ShatteredEngine_Testing "ShatteredEngine" "SDL2" "SDL2main" ${macOS_Libraries})
endif()
if(_SHATTERED_UNIX)
    target_link_libraries(ShatteredEngine_Testing PUBLIC "ShatteredEngine" PUBLIC "SDL2-static" PUBLIC "SDL2main" PUBLIC "X11" PUBLIC "GL" PUBLIC "dl" PUBLIC "rt" PUBLIC "pthread")
    if(_SHATTERED_VULKAN)
        target_link_libraries(ShatteredEngine_Testing PUBLIC "vulkan")
    endif()
endif()
